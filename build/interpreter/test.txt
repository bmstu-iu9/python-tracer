def hex(digit):
    if digit <= 9:
        digit = str(digit)
    elif digit <= 13:
        if digit <= 11:
            if digit == 10:
                digit = 'A'
            else:
                digit = 'B'
        elif digit == 12:
            digit = 'C'
        else:
            digit = 'D'
    elif digit == 14:
        digit = 'E'
    else:
        digit = 'F'
    return digit

res = hex(15)
